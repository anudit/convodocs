import Markdown from 'markdown-to-jsx';
import { useSSG } from 'nextra/ssg';
import glossary from "../../public/glossary/glossary.json";

export const GlossaryRender = () => {
    const mdData = Object.entries(glossary).map(([word, meaning])=>{
        return `### ${word} \n ${meaning.slice(0, 1).toUpperCase()}${meaning.slice(1)}`;
    }).join('\n');

    return(
        <Markdown id="glossary">
            { mdData }
        </Markdown>
    )
}
