import CommentBox from './../../../components/CommentBox';
import Markdown from 'markdown-to-jsx';
import { useSSG } from 'nextra/ssg';

export const getStaticProps = async ({ params }) => {

  let promiseArray = [
    fetch(`https://raw.githubusercontent.com/anudit/convosdk/main/packages/sdk/CHANGELOG.md`).then(response => response.text()),
    fetch(`https://raw.githubusercontent.com/anudit/convosdk/main/packages/react/CHANGELOG.md`).then(response => response.text())
  ];

  let res = await Promise.allSettled(promiseArray);
  let sdkChangelog= res[0].value.split('\n').slice(1).join('\n');
  let reactChangelog= res[1].value.split('\n').slice(1).join('\n');

  return ({ props: { ssg: {sdkChangelog, reactChangelog} }, revalidate: 10 })

}

export const SdkChangelogRender = () => {
  const {sdkChangelog} = useSSG();
  return (
      <Markdown>
        {sdkChangelog}
      </Markdown>
  )
}

export const ReactChangelogRender = () => {
  const { reactChangelog} = useSSG();
  return (
    <Markdown>
        {reactChangelog}
      </Markdown>
  )
}
