import CommentBox from './../../../../components/CommentBox';
import Markdown from 'markdown-to-jsx';
import { useSSG } from 'nextra/ssg';

export const getStaticProps = async ({ params }) => {

  let log = await fetch(`https://raw.githubusercontent.com/anudit/convosdk/main/packages/react/CHANGELOG.md`).then(response => response.text());
  log = log.split('\n').slice(1).join('\n');

  return ({ props: { ssg: {log} }, revalidate: 10 })

}

export const LogRender = () => {
  const {log} = useSSG();
  return (
      <Markdown>
        {log}
      </Markdown>
  )
}
